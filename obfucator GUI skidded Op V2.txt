-- Obfuscation Function
local function obfuscateScript(scriptText)
    local encoded = {}
    for i = 1, #scriptText do
        table.insert(encoded, tostring(string.byte(scriptText, i)))
    end
    return "local obfuscated_script = {" .. table.concat(encoded, ",") .. "}\nlocal decoded = ''\nfor i = 1, #obfuscated_script do decoded = decoded .. string.char(obfuscated_script[i]) end\nloadstring(decoded)()"
end

-- Button Click Function
button.MouseButton1Click:Connect(function()
    textbox.Text = obfuscateScript(textbox.Text)
end)
